//looping through 2d array

int main(){
    int ages[2][3] = {{1,2,3},{4,5,6}};
    int i,j;
    for(i=0;i<2;i++){
        for(j=0;j<3;j++){
            printf("%d ", ages[i][j]);
        }
    }
    return 0;
}
